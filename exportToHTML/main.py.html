a = input('- Weight(in Kilogram) : ')
b = input('- Height(in Meters) : ')
bmi = float(a)/float(b)**2
if bmi < 18.5:
    print('- You are Underweight.')
elif 18.5 < bmi < 24.9:
    print('- Your are Normal.')
elif 25 < bmi < 29.9:
    print('- Your are Overweigt.')
elif 30 < bmi:
    print('- You are obese.')
print('- Your BMI is ',bmi)
